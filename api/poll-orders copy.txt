// - DMOS_QUEUE_MAX_PER_RUN (default 5)

const { Redis } = require("@upstash/redis");
const redis = Redis.fromEnv();

function json(res, status, data) {
  res.statusCode = status;
  res.setHeader("Content-Type", "application/json");
  res.end(JSON.stringify(data, null, 2));
}

function isAuthorized(req) {
  const secret = req.query && req.query.secret;
  const expected = process.env.CRON_SECRET;
  if (!expected) return { ok: false, error: "Missing CRON_SECRET env var" };
  if (!secret) return { ok: false, error: "Missing ?secret=" };
  if (secret !== expected) return { ok: false, error: "Unauthorized" };
  return { ok: true };
}

function iso(d) {
  return d.toISOString();
}

function safeDate(value) {
  const d = new Date(value);
  if (Number.isNaN(d.getTime())) return null;
  return d;
}

async function fetchSquarespaceOrders({ modifiedAfter, modifiedBefore }) {
  const apiKey = process.env.SQUARESPACE_API_KEY;
  if (!apiKey) {
    return {
      ok: false,
      where: "env",
      status: 500,
      body: "Missing SQUARESPACE_API_KEY env var",
    };
  }

  const base = "https://api.squarespace.com/1.0/commerce/orders";

  const url = new URL(base);
  url.searchParams.set("modifiedAfter", iso(modifiedAfter));
  url.searchParams.set("modifiedBefore", iso(modifiedBefore));
  url.searchParams.set("limit", "50");

  const resp = await fetch(url.toString(), {
    method: "GET",
    headers: {
      Authorization: `Bearer ${apiKey}`,
      "Content-Type": "application/json",
      "User-Agent": "DMOS Order Router (domoreonshore.com)",
    },
  });

  const text = await resp.text();

  // Minimal debug log (no PII)
  console.log("Squarespace list orders response:", {
    status: resp.status,
    preview: text.slice(0, 200),
  });

  if (!resp.ok) {
    return { ok: false, where: "fetch_orders", status: resp.status, body: text };
  }

  let data = null;
  try {
    data = JSON.parse(text);
  } catch (e) {
    return { ok: false, where: "parse_orders", status: 500, body: text.slice(0, 500) };
  }

  const orders = Array.isArray(data) ? data : data.result || data.orders || [];
  return { ok: true, orders };
}

async function fetchOrderDetail(orderId) {
  const apiKey = process.env.SQUARESPACE_API_KEY;
  const url = `https://api.squarespace.com/1.0/commerce/orders/${orderId}`;

  const resp = await fetch(url, {
    method: "GET",
    headers: {
      Authorization: `Bearer ${apiKey}`,
      "Content-Type": "application/json",
      "User-Agent": "DMOS Order Router (domoreonshore.com)",
    },
  });

  const text = await resp.text();

  if (!resp.ok) {
    return { ok: false, status: resp.status, body: text };
  }

  try {
    return { ok: true, data: JSON.parse(text) };
  } catch (e) {
    return { ok: false, status: 500, body: text.slice(0, 500) };
  }
}

/**
 * Payment determination:
 * Squarespace fields can vary. We try common ones.
 * Also supports "free order" where grandTotal is 0.00 due to discounts.
 */
function isPaidOrFree(order) {
  const s = String(
    order?.paymentStatus ||
      order?.orderStatus ||
      order?.status ||
      order?.fulfillmentStatus ||
      ""
  ).toUpperCase();

  if (s.includes("PAID")) return true;
  if (s.includes("FULFILL")) return true;

  // Look for successful transactions
  if (Array.isArray(order?.transactions)) {
    const okTxn = order.transactions.some((t) =>
      String(t?.status || "").toUpperCase().includes("SUCC")
    );
    if (okTxn) return true;
  }

  // Handle promo/free order (grandTotal 0.00)
  const grand = Number(order?.grandTotal?.value || order?.grandTotal || 0);
  const subtotal = Number(order?.subtotal?.value || order?.subtotal || 0);
  const discount = Number(order?.discountTotal?.value || order?.discountTotal || 0);
  const refunded = Number(order?.refundedTotal?.value || order?.refundedTotal || 0);

  // If it’s a legitimate free order: subtotal > 0, grandTotal == 0, discount == subtotal, no refunds.
  if (subtotal > 0 && grand === 0 && discount >= subtotal && refunded === 0) return true;

  return false;
}

function buildLuluPayloadFromOrder(order) {
  const shipping = order?.shippingAddress || {};

  // Validate shipping address essentials (Lulu needs these)
  const required = ["address1", "city", "state", "postalCode", "countryCode"];
  for (const k of required) {
    if (!shipping?.[k]) {
      return { ok: false, error: `Missing shippingAddress.${k}` };
    }
  }

  const lineItems = Array.isArray(order?.lineItems) ? order.lineItems : [];
  const printableItems = [];

  for (const li of lineItems) {
    // Only print physical products
    if (li?.lineItemType !== "PHYSICAL_PRODUCT") continue;

    const sku = li?.sku;
    const qty = Number(li?.quantity || 0);
    if (!sku || qty <= 0) continue;

    const mapping = SKU_TO_LULU[sku];
    if (!mapping) {
      // Unknown SKU: ignore for now (could be digital upsell later)
      continue;
    }

    // Ensure URLs are real before trying to print
    if (
      !mapping.interiorPdfUrl ||
      !mapping.coverPdfUrl ||
      mapping.interiorPdfUrl.includes("YOUR_DOMAIN_OR_CDN") ||
      mapping.coverPdfUrl.includes("YOUR_DOMAIN_OR_CDN")
    ) {
      return { ok: false, error: `SKU ${sku} is missing real PDF URLs` };
    }

    printableItems.push({
      sku,
      quantity: qty,
      title: mapping.title,
      interiorPdfUrl: mapping.interiorPdfUrl,
      coverPdfUrl: mapping.coverPdfUrl,
    });
  }

  if (printableItems.length === 0) {
    return { ok: false, error: "No printable PHYSICAL_PRODUCT SKUs found" };
  }

  // Lulu payload structure can vary by product workflow.
  // This is a reasonable starting shape.
  const payload = {
    contact_email: order?.customerEmail || "orders@domoreonshore.com",
    shipping_address: {
      name: shipping?.name || "Customer",
      street1: shipping.address1,
      street2: shipping.address2 || "",
      city: shipping.city,
      state_code: shipping.state,
      postcode: shipping.postalCode,
      country_code: shipping.countryCode,
      phone_number: shipping.phone || "",
    },
    external_id: order?.id,
    line_items: printableItems.map((p) => ({
      quantity: p.quantity,
      title: p.title,
      print_job_files: {
        interior: { source_url: p.interiorPdfUrl },
        cover: { source_url: p.coverPdfUrl },
      },
    })),
  };

  return { ok: true, payload, printableItems };
}

async function getLuluAccessToken() {
  const key = process.env.LULU_CLIENT_KEY;
  const secret = process.env.LULU_CLIENT_SECRET;
  if (!key || !secret) return { ok: false, error: "Missing Lulu credentials" };

  const basicAuth = Buffer.from(`${key}:${secret}`).toString("base64");

  const resp = await fetch(
    "https://api.lulu.com/auth/realms/glasstree/protocol/openid-connect/token",
    {
      method: "POST",
      headers: {
        Authorization: `Basic ${basicAuth}`,
        "Content-Type": "application/x-www-form-urlencoded",
      },
      body: "grant_type=client_credentials",
    }
  );

  const data = await resp.json();
  if (!resp.ok) return { ok: false, error: "Lulu auth failed", details: data };
  return { ok: true, token: data.access_token };
}

async function createLuluPrintJob(accessToken, luluPayload) {
  const resp = await fetch("https://api.lulu.com/print-jobs/", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${accessToken}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify(luluPayload),
  });

  const data = await resp.json();
  if (!resp.ok) return { ok: false, status: resp.status, details: data };
  return { ok: true, data };
}

module.exports = async (req, res) => {
  try {
    if (req.method !== "GET") {
      res.setHeader("Allow", "GET");
      return json(res, 405, { ok: false, error: "Method Not Allowed" });
    }

    const auth = isAuthorized(req);
    if (!auth.ok) return json(res, 401, { ok: false, error: auth.error });

    const debug = req.query && req.query.debug === "1";

    // DRY_RUN defaults to "1" unless you explicitly set it to "0"
    const dryRunEnv = process.env.DMOS_DRY_RUN;
    const DRY_RUN = dryRunEnv == null ? true : String(dryRunEnv) !== "0";

    const MAX_PER_RUN = Number(process.env.DMOS_QUEUE_MAX_PER_RUN || 5);

    const CURSOR_KEY = "dmos:lastModifiedAfter";
    const QUEUE_KEY = "dmos:printQueue"; // Redis set of order IDs to attempt printing

    const last = await redis.get(CURSOR_KEY);

    const now = new Date();

    // Start with last cursor if present, else last 24h for safety
    const fallbackStart = new Date(now.getTime() - 24 * 60 * 60 * 1000);
    const stored = null ? safeDate(last) : null;

    // Overlap to avoid boundary misses
    const overlapMs = 2 * 60 * 1000;

    const modifiedAfter = stored
      ? new Date(Math.max(stored.getTime() - overlapMs, 0))
      : fallbackStart;

    const modifiedBefore = now;

    const fetched = await fetchSquarespaceOrders({ modifiedAfter, modifiedBefore });

    if (!fetched.ok) {
      return json(res, fetched.status || 500, {
        ok: false,
        where: fetched.where,
        status: fetched.status,
        bodyPreview: typeof fetched.body === "string" ? fetched.body.slice(0, 800) : "",
      });
    }

    const orders = fetched.orders || [];

    // Add all seen order IDs to the print queue unless already printed
    let queuedAdded = 0;
    for (const o of orders) {
      const id = o.id || o.orderId;
      if (!id) continue;

      const printedKey = `dmos:printed:${id}`;
      const alreadyPrinted = await redis.get(printedKey);
      if (alreadyPrinted) continue;

      await redis.sadd(QUEUE_KEY, id);
      queuedAdded += 1;
    }

    // Update cursor immediately (we won’t lose orders because the queue holds them)
    await redis.set(CURSOR_KEY, iso(modifiedBefore));

    // Pull queue members and attempt up to MAX_PER_RUN
    const queueMembers = (await redis.smembers(QUEUE_KEY)) || [];
    const toAttempt = queueMembers.slice(0, MAX_PER_RUN);

    // If dry run, we do not auth to Lulu or create print jobs
    let luluToken = null;
    if (!DRY_RUN && toAttempt.length > 0) {
      const luluAuth = await getLuluAccessToken();
      if (!luluAuth.ok) {
        return json(res, 500, { ok: false, error: luluAuth.error, details: luluAuth.details });
      }
      luluToken = luluAuth.token;
    }

    const printed = [];
    const skipped = [];
    const failed = [];

    for (const orderId of toAttempt) {
      const printedKey = `dmos:printed:${orderId}`;
      const alreadyPrinted = await redis.get(printedKey);
      if (alreadyPrinted) {
        await redis.srem(QUEUE_KEY, orderId);
        skipped.push({ id: orderId, reason: "already_printed" });
        continue;
      }

      const detail = await fetchOrderDetail(orderId);
      if (!detail.ok) {
        failed.push({ id: orderId, reason: "detail_fetch_failed", status: detail.status });
        continue; // leave in queue for retry
      }

      const order = detail.data;

      // Only print paid/free orders
      if (!isPaidOrFree(order)) {
        // Not paid yet: keep in queue for now (or you can choose to remove)
        skipped.push({ id: orderId, reason: "not_paid_yet" });
        continue;
      }

      // Build Lulu payload (and validate PDFs are present)
      const built = buildLuluPayloadFromOrder(order);
      if (!built.ok) {
        // If PDFs aren’t ready yet, this will show up here.
        skipped.push({ id: orderId, reason: built.error });
        continue;
      }

      if (DRY_RUN) {
        // Do NOT print. Just report what would have printed.
        printed.push({
          id: orderId,
          dryRun: true,
          items: built.printableItems.map((p) => ({ sku: p.sku, quantity: p.quantity })),
        });

        // Do NOT mark printed in dry run, and do NOT remove from queue unless you want to.
        // Leaving in queue is safer while testing.
        continue;
      }

      const created = await createLuluPrintJob(luluToken, built.payload);
      if (!created.ok) {
        failed.push({
          id: orderId,
          reason: "lulu_create_failed",
          status: created.status,
          details: created.details,
        });
        continue; // leave in queue for retry
      }

      // Success: mark printed and remove from queue
      await redis.set(printedKey, iso(new Date()));
      await redis.srem(QUEUE_KEY, orderId);

      printed.push({
        id: orderId,
        luluPrintJobId: created.data?.id || created.data?.print_job_id || null,
        items: built.printableItems.map((p) => ({ sku: p.sku, quantity: p.quantity })),
      });
    }

    // Safe summary logs (no email/address)
    console.log("DMOS poll result:", {
      fetched: orders.length,
      queueSize: queueMembers.length,
      attempted: toAttempt.length,
      queuedAdded,
      dryRun: DRY_RUN,
      printedCount: printed.length,
      skippedCount: skipped.length,
      failedCount: failed.length,
      printedPreview: printed.slice(0, 3),
      skippedPreview: skipped.slice(0, 3),
      failedPreview: failed.slice(0, 1),
    });

    return json(res, 200, {
      ok: true,
      dryRun: DRY_RUN,
      checkedAfter: iso(modifiedAfter),
      checkedBefore: iso(modifiedBefore),
      fetched: orders.length,
      queuedAdded,
      queueSize: queueMembers.length,
      attempted: toAttempt.length,
      printed,
      skipped,
      failed,
      ...(debug ? { debug: { cursorStored: last || null, attemptedIds: toAttempt } } : {}),
    });
  } catch (err) {
    console.error("DMOS poll error:", err);
    return json(res, 500, { ok: false, error: err?.message || "Server error" });
  }
};
